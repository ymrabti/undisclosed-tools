var one_sec =1000;var Mode12 = false;

function Update(time){
    var date = time? new Date(time):new Date();
    var hrsutc = ((-1)*date.getTimezoneOffset()/60).toFixed(2);
    Citypara = `(GMT${hrsutc!=0 ? ` ${hrsutc>0?`+${hrsutc}`:hrsutc} H`:``})`;
    AngleSecnds = date.getSeconds() * 6;
    AngleMinuts = date.getMinutes() * 6;
    AngleHourss = date.getHours() * 30 + .5 * date.getMinutes();
    readAngles(Mode12);
    MoveSecnds(0);MoveMinuts(0);MoveHourss(0);
    /* Update every seconde */
}
function readAngles(M12){Mode12=M12;
    MoveSecnds();
    Numeric.innerText = getNumeric(AngleHourss,AngleMinuts,AngleSecnds,M12);
    var secnd = (AngleSecnds%360)/6;
    if (!secnd) {MoveMinuts();MoveHourss();}
    Numeric.innerText = getNumeric(AngleHourss,AngleMinuts,AngleSecnds,M12);
}

function getNumeric(Ah,Am,As,M12) {
    var mod = Ah%30;

    var minte = mod*2;var mnte1 = (Am%360)/6;
    var heure = ((Ah%720-mod)/30)%24;

}
